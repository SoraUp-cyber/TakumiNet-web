public class PlayerController : MonoBehaviour {
    public float speed = 5.0f;
    public float jumpForce = 10.0f;

    private Rigidbody rb;
    private bool isGrounded = false;

    void Start() {
        rb = GetComponent(); // Obtiene el componente Rigidbody del objeto
    }

    void Update() {
        // Movimiento
        float horizontalInput = Input.GetAxis("Horizontal");
        float verticalInput = Input.GetAxis("Vertical");
        Vector3 movement = new Vector3(horizontalInput, 0, verticalInput);
        movement = movement.normalized * speed * Time.deltaTime;
        rb.MovePosition(transform.position + movement);

        // Salto
        if (Input.GetButtonDown("Jump") && isGrounded) {
            rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
            isGrounded = false; // El personaje ya no est√° en el suelo
        }

        // Detectar si el personaje est√° en el suelo
        if (rb.GetComponent<Collider>().bounds.Intersects(new Bounds(Vector3.zero, new Vector3(1f, 0.1f, 1f)))) {
            isGrounded = true;
        }
    }
}
